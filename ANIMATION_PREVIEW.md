### パフォーマンス最適化

```css
/* GPU アクセラレーション */
.stage {
  transform: translateZ(0);
  will-change: opacity, transform;
}

/* スムーズなアニメーション */
.plant {
  transition: all 0.5s cubic-bezier(0.4, 0, 0.2, 1);
}
```

### アニメーション実装

```javascript
// ステージ管理
let currentStage = 1;
const totalStages = 4;

// 自動進行
setInterval(() => {
  nextStage();
}, stageDuration);

// スムーズなトランジション
function nextStage() {
  // フェードアウト
  currentElement.style.opacity = 0;
  
  // 次のステージにフェードイン
  setTimeout(() => {
    nextElement.style.opacity = 1;
  }, 300);
}
```
